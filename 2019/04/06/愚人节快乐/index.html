And then output the Wechat

题解

  • 因为题设确保一定有解且解一定是两个质因数，所以我们不用验证是不是质数也不用考虑质因数个数问题
  • 直接用质因数迭代求最大质因数就行你看脱单就是这么简单（雾

Source Code

#include <iostream>
#define ll long long int

using namespace std;

char num[20];
ll arr[20];

//求最大质因数
ll GetMaxPrime(ll n){
    ll i = 2;
    ll res = 1;
    while (n > 2){
        if (n % i == 0){
            n /= i;
            res = i;
        } else
            i++;
    }
    return res;
}

void data(char num[]){
    ll integer = 0, mod = 0, res;
    int n = 0;
    while (num[n] != '*') n++;


    for (int j = 0; j < 10 - n; j++){
        int t = j + 1;
        mod = num[j] - '0';
        while (t < 10){
            mod *= 10;
            t++;
        }
        integer += mod;
    }

    for (int k = 1; k < 1000; k++){
        integer += k;
        res = integer / GetMaxPrime(integer);
        if (res == GetMaxPrime(res) ){
            cout << res << endl;
            integer /= res;
            cout << integer << endl;
            return;
        }

    }


}


int main(){
